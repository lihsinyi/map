169  void runScheduler_()
170  {
171  // Start simulation
172  sparta_scheduler_->run(1, true);
173  sparta_assert(sparta_scheduler_->getCurrentTick() == 0);
174  sparta_assert(sparta_scheduler_->nextEventTick() > 0);
175 
176  do {
177 
178  // Wait for the next clock edge
179  // Wait for either the next time slice. TODO: Add an event
180  // to wake the scheduler earlier if need be
181  wait(sc_core::sc_time(double(sparta_scheduler_->nextEventTick() -
182  sc_core::sc_time_stamp().value()), sparta_sc_time_));
183 
184  // Align to the posedge events in systemc
185  wait(sc_core::SC_ZERO_TIME);
186 
187  // If given, schedule the user's SystemC query event to
188  // allow the Sparta user to check to see if the SystemC side
189  // of simulation is complete. This query can be as simple
190  // as asking the SystemC kernel if there are any events
191  // (sc_core::sc_pending_activity()) or asking the SystemC
192  // simulator if it's done.
193  if(sysc_query_event_ && sparta_scheduler_->getCurrentTick() >= next_sysc_event_fire_tick_)
194  {
195  next_sysc_event_fire_tick_ = sparta_scheduler_->getCurrentTick() + sysc_query_event_interval_;
196  sysc_query_event_->scheduleRelativeTick(1, sparta_scheduler_);
197  }
198 
199  advanceSpartaScheduler_();
200 
201  } while(!sparta_scheduler_->isFinished() || !sysc_simulation_done_);
202 
203  // Stop simulation
204  sc_core::sc_stop();
205  sc_stop_called_ = true;
206  }

208  void advanceSpartaScheduler_()
209  {
210  sc_core::sc_time sysc_time = sc_core::sc_time_stamp();
211 
212  // The SystemC scheduler will always lead or be exactly at
213  // the same tick as Sparta. Following this rule allows safe
214  // assumptions in scheduling synchronization.
215  //
216  // If the Sparta Scheduler next tick is @ 1000 and we're at
217  // 500, we want to advance 500 ticks and end @ 1000. The
218  // next event tick time should never be the same as
219  // current tick unless it's the beginning of simulation.
220  if(sysc_time.value() > sparta_scheduler_->getCurrentTick()) {
221  const bool exacting_run = true;
222  const bool measure_scheduler_time = false; // no need to do this
223  sparta_scheduler_->run(sysc_time.value() - sparta_scheduler_->getCurrentTick(),
224  exacting_run, measure_scheduler_time);
225  }
226 
227  // Make sure we're in sync
228  sparta_assert(sysc_time.value() == sparta_scheduler_->getCurrentTick());
229  }
