599  void erase(const uint32_t& idx)
600  {
601  // Make sure we are invalidating an already valid object.
602  sparta_assert(idx < size(), "Cannot erase an index that is not already valid");
603  // Do the invalidation immediately
604  // 1. Move the free space pointer to the erased position.
605  // 2. Set the current free space pointer's next to the old free position
606  DataPointer* oldFree = free_position_;
607  free_position_ = buffer_map_[idx];
608  free_position_->next_free = oldFree;
609 
610  // Mark DataPointer as invalid
611  validator_->detachDataPointer(free_position_);
612 
613  // Shift all the positions above the invalidation in the map one space down.
614  uint32_t i = idx;
615  sparta_assert(num_valid_ > 0);
616  const uint32_t top_idx_of_buffer = num_valid_ - 1;
617  while(i < top_idx_of_buffer)
618  {
619  // assert that we are not going to do an invalid read.
620  sparta_assert(i + 1 < num_entries_);
621  buffer_map_[i] = buffer_map_[i + 1];
622  buffer_map_[i]->physical_idx = i;
623 
624  // Shift the indexes in the address map.
625  address_map_[i] = address_map_[i + 1];
626  ++i;
627  }
628  // the entry at the old num_valid_ in the map now points to nullptr
629  buffer_map_[top_idx_of_buffer] = nullptr;
630 
631  // Remove this entry of the address map as it becomes a free position.
632  address_map_.erase(top_idx_of_buffer);
633  // update counts.
634  --num_valid_;
635  updateUtilizationCounters_();
636 
637  }

840  void resizeInternalContainers_() {
841 
842  // Assert that the Buffer class is in Infinite-Mode.
843  sparta_assert(is_infinite_mode_, "The Buffer class must be in Infinite-Mode in order to resize itself.");
844 
845  // We do not resize if there are available slots in buffer.
846  if(numFree() != 0) {
847  return;
848  }
849 
850  // Resize the buffer_map_ with the amount provided by user.
851  buffer_map_.resize(buffer_map_.capacity() + resize_delta_);
852 
853  // The number of entries the buffer can hold is its capacity.
854  num_entries_ = buffer_map_.capacity();
855 
856  // Resize the data_pool_ to twice the capacity of the buffer_map_.
857  data_pool_.resize(num_entries_ * 2);
858 
859  // The number of entries the pool can hold is its capacity.
860  data_pool_size_ = data_pool_.capacity();
861 
862 
863  // Each entry in data_pool_ should have their next free position
864  // pointer point to the slot to its right.
865  for(uint32_t i = 0; i < data_pool_size_ - 1; ++i) {
866  data_pool_[i].next_free = &data_pool_[i + 1];
867  }
868 
869  // The last entry should point to itself as the next free position.
870  data_pool_[data_pool_size_ - 1].next_free = &data_pool_[data_pool_size_ - 1];
871 
872  // The first position should point to the first entry.
873  first_position_ = &data_pool_[0];
874 
875  // The free position should point to the location according to
876  // the number of entries in the buffer.
877  free_position_ = &data_pool_[num_valid_];
878 
879  // Make all the pointers in buffer_map_ point to the appropriate indexes.
880  for(uint32_t i = 0; i < num_valid_; ++i) {
881  buffer_map_[i] = &data_pool_[address_map_[i]];
882  }
883 
884  // Resize the validator vector and relink the validator data pool.
885  validator_->resizeIteratorValidator(resize_delta_, data_pool_);
886  }
